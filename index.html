Connect your Google Sheet
Uses a Google Apps Script webhook — no OAuth tokens, no expiry, works forever automatically. Do this once and every order syncs to your sheet instantly.
1
Create a Google Sheet and open Extensions → Apps Script
Go to sheets.new to create a fresh sheet, then click Extensions in the menu bar → Apps Script.
2
Delete everything in the editor, paste the code below, then click Deploy → New deployment → Web app
function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    if (data.type === 'order') {
      var sheet = ss.getSheetByName('Orders') || ss.insertSheet('Orders');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Date','Order #','Channel','Qty','Unit Price','Total','Stock After','Notes']);
        sheet.getRange(1,1,1,8).setFontWeight('bold').setBackground('#0e0c0a').setFontColor('#f0c060');
      }
      sheet.appendRow([data.date, data.num, data.chan, data.qty, data.price, data.total, data.stockAfter, data.notes]);
    }

    if (data.type === 'consignment') {
      var cs = ss.getSheetByName('Consignment') || ss.insertSheet('Consignment');
      if (cs.getLastRow() === 0) {
        cs.appendRow(['Date','Store','Event','Qty','Rate %','Amount Due','Notes','Status']);
        cs.getRange(1,1,1,8).setFontWeight('bold').setBackground('#0e0c0a').setFontColor('#f0c060');
      }
      cs.appendRow([data.date, data.store, data.event, data.qty, data.rate, data.amountDue, data.notes, data.status]);
    }

    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false, error:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
When the Deploy dialog opens: set Execute as → Me and Who has access → Anyone. Click Deploy and copy the Web App URL.
