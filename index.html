Connect your Google Sheet
Uses a Google Apps Script webhook — paste your Web App URL once and every order syncs automatically. One connection covers all books.
1
Create a Google Sheet → Extensions → Apps Script
Go to sheets.new, then Extensions → Apps Script.
2
Paste this code, deploy as Web App (Execute as: Me, Who has access: Anyone)
function doPost(e) {
  try {
    var data = JSON.parse(e.parameter.payload);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var bookName = data.book || 'Unknown';

    if (data.type === 'order') {
      var sheetName = bookName + ' — Orders';
      var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName);
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Date','Order #','Channel','Qty','Unit Price','Total','Stock After','Notes']);
        sheet.getRange(1,1,1,8).setFontWeight('bold').setBackground('#0e0c0a').setFontColor('#f0c060');
      }
      sheet.appendRow([data.date, data.num, data.chan, data.qty, data.price, data.total, data.stockAfter, data.notes]);
    }

    if (data.type === 'consignment') {
      var csName = bookName + ' — Consignment';
      var cs = ss.getSheetByName(csName) || ss.insertSheet(csName);
      if (cs.getLastRow() === 0) {
        cs.appendRow(['Date','Store','Event','Qty','Rate %','Amount Due','Notes','Status']);
        cs.getRange(1,1,1,8).setFontWeight('bold').setBackground('#0e0c0a').setFontColor('#f0c060');
      }
      cs.appendRow([data.date, data.store, data.event, data.qty, data.rate, data.amountDue, data.notes, data.status]);
    }

    return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({ok:false,error:err.message})).setMimeType(ContentService.MimeType.JSON);
  }
}
The script creates a separate sheet tab per book automatically (e.g. "Un Fantastico Altrove — Orders" and "The Hound — Orders").
